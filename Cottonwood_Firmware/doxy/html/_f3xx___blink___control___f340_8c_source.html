171 /* Initialize system clock to maximum frequency. */

173 /* ---------------------------------------------------------------------------- */
174 void Sysclk_Init(void)
175 {
176 #ifdef _USB_LOW_SPEED_
177 
178  OSCICN |= 0x03; /* Configure internal oscillator for */
179  /* its maximum frequency and enable */
180  /* missing clock detector */
181 
182  CLKSEL = SYS_INT_OSC; /* Select System clock */
183  CLKSEL |= USB_INT_OSC_DIV_2; /* Select USB clock */
184 #else
185 
186 #if (SYS_CLK_SELECT == SYS_USE_CLK_INT)
187  OSCICN |= 0x03; /* Configure internal oscillator for */
188  /* its maximum frequency and enable */
189  /* missing clock detector */
190 
191  CLKMUL = 0x00; /* Select internal oscillator as */
192  /* input to clock multiplier */
193 
194  CLKMUL |= 0x80; /* Enable clock multiplier */
195  Delay(); /* Delay for clock multiplier to begin */
196  CLKMUL |= 0xC0; /* Initialize the clock multiplier */
197  Delay(); /* Delay for clock multiplier to begin */
198 
199  while (!(CLKMUL & 0x20)); /* Wait for multiplier to lock */
200 #if (SYSCLK == 48000000)
201  FLSCL = 0x90; /* Set flash waitstates for freqs > 25 MHz. */
202  CLKSEL = SYS_4X | USB_4X_CLOCK; /* Set system clock to 48MHz */
203 #elif (SYSCLK == 24000000)
204  CLKSEL = SYS_4X_DIV_2 | USB_4X_CLOCK; /* Set system clock to 24MHz */
205 #elif (SYSCLK == 12000000)
206  CLKSEL = SYS_INT_OSC | USB_4X_CLOCK; /* Set system clock to 24MHz */
207 #endif
208 
209 #elif (SYS_CLK_SELECT == SYS_USE_CLK_EXT_12MHZ)
210  int cnt;
211 
212  OSCXCN = 0x20; /* set XOSCMD to 010b External CMOS Osc Mode */
213  OSCXCN |= 0x07; /* set XFCN to 111b Ext. Osc. Freq. */
214 
215  /* wait >= 1ms */
216  for (cnt = 0; cnt < 20; cnt++)
217  {
218  Delay(); /* Delay for external oscillator becoming stable */
219  }
220 
221  CLKMUL = 0x00; /* Reset clock multiplier */
222  CLKMUL = 0x01; /* Select external oscillator as */
223  /* input to clock multiplier */
224 
225  CLKMUL |= 0x80; /* Enable clock multiplier */
226  Delay(); /* Delay for clock multiplier to begin */
227  CLKMUL |= 0xC0; /* Initialize the clock multiplier */
228  Delay(); /* Delay for clock multiplier to begin */
229 
230  while (!(CLKMUL & 0x20)); /* Wait for multiplier to lock */
231 #if (SYSCLK == 48000000)
232  FLSCL = 0x90; /* Set flash waitstates for freqs > 25 MHz. */
233  CLKSEL = SYS_4X | USB_4X_CLOCK; /* Set system clock to 48MHz */
234 #elif (SYSCLK == 24000000)
235  CLKSEL = SYS_4X_DIV_2 | USB_4X_CLOCK; /* Set system clock to 24MHz */
236 #elif (SYSCLK == 12000000)
237  CLKSEL = SYS_INT_OSC | USB_4X_CLOCK; /* Set system clock to 24MHz */
238 #endif
239 
240  OSCICN &= ~0x80; /* explicitly switch off internal H-F oscillator */
241  OSCLCN &= ~0x80; /* explicitly switch off internal L-F oscillator */
242 
243 
244 #elif (SYS_CLK_SELECT == SYS_USE_CLK_EXT_48MHZ)
245  int cnt;
246 
247  OSCXCN = 0x20; /* set XOSCMD to 010b External CMOS Osc Mode */
248  OSCXCN |= 0x07; /* set XFCN to 111b Ext. Osc. Freq. */
249 
250  /* wait >= 1ms */
251  for (cnt = 0; cnt < 20; cnt++)
252  {
253  Delay(); /* Delay for external oscillator becoming stable */

255

269 #endif

274 /*----------------------------------------------------------------------------- */
275 /* USB0_Init */
276 /*----------------------------------------------------------------------------- */
277 /* */
278 /* Return Value - None */
279 /* Parameters - None */
280 /* */
281 /* - Initialize USB0 */
282 /* - Enable USB0 interrupts */
283 /* - Enable USB0 transceiver */
284 /* - Enable USB0 with suspend detection */
285 /* */
286 /* ---------------------------------------------------------------------------- */
287 void Usb_Init(void)
288 {
289  POLL_WRITE_BYTE(POWER, 0x08); /* Force Asynchronous USB Reset */
290  POLL_WRITE_BYTE(IN1IE, 0x07); /* Enable Endpoint 0-2 in interrupts */
291  POLL_WRITE_BYTE(OUT1IE, 0x07); /* Enable Endpoint 0-2 out interrupts */
292  POLL_WRITE_BYTE(CMIE, 0x07); /* Enable Reset, Resume, and Suspend */
293  /* interrupts */
294 #ifdef _USB_LOW_SPEED_
295  USB0XCN = 0xC0; /* Enable transceiver; select low speed */
296  POLL_WRITE_BYTE(CLKREC, 0xA0); /* Enable clock recovery; single-step */
297  /* mode disabled; low speed mode enabled */
298 #else
299  USB0XCN = 0xE0; /* Enable transceiver; select full speed */
300  POLL_WRITE_BYTE(CLKREC, 0x80); /* Enable clock recovery, single-step */
301  /* mode disabled */
302 #endif /* _USB_LOW_SPEED_ */
303 
304  EIE1 |= 0x02; /* Enable USB0 Interrupts */
305  EA = 1; /* Global Interrupt enable */
306  /* Enable USB0 by clearing the USB */
307  /* Inhibit bit */
308  POLL_WRITE_BYTE(POWER, 0x01); /* and enable suspend detection */
309 }
