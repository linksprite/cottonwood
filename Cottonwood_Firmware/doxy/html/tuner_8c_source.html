118 void tunerSetCap( u8 component, u8 val )
119 {
120  u8 d;
121  u8 spi0cfg, spi0ckr;
122 
123  /* save spi defaults */
124  spi0cfg = SPI0CFG;
125  spi0ckr = SPI0CKR;
126 
127  /* reconfigure spi */
128  SPI0CFG = 0x00; /* off */
129 #if (CLK == 48000000) /* reg = f_sys/(2f) - 1 */
130  SPI0CKR = 48/(1*2)-1; /* 1 MHz */
131 #elif (CLK == 24000000)
132  SPI0CKR = 24/(1*2)-1; /* 1 MHz */
133 #elif (CLK == 12000000)
134  SPI0CKR = 12/(1*2)-1; /* 1 MHz */
135 #endif
136  SPI0CFG = 0x40;/* MSTEN=1, CPHA=0, CKPOL=0, */
137 
138  switch (component)
139  {
140  case TUNER_Cin:
141  SEN_TUNER_CIN(1);
142  break;
143  case TUNER_Clen:
144  SEN_TUNER_CLEN(1);
145  break;
146  case TUNER_Cout:
147  SEN_TUNER_COUT(1);
148  break;
149  }
150 
151  SPI0DAT = val;
152  while(!(SPI0CN & 0x80));
153  SPI0CN = 1; /* Clear the interrupt flag */
154  d = SPI0DAT; /* dummy receive this byte */
155 
156  switch (component)
157  {
158  case TUNER_Cin:
159  SEN_TUNER_CIN(0);
160  break;
161  case TUNER_Clen:
162  SEN_TUNER_CLEN(0);
163  break;
164  case TUNER_Cout:
165  SEN_TUNER_COUT(0);
166  break;
167  }
168  /* restore spi to default */
169  SPI0CFG = spi0cfg;
170  SPI0CKR = spi0ckr;
171 }
172 u16 tunerGetReflected(void)
173 {
174  u16 r; u32 refl;
175  s8 i, q;
176 #ifdef POWER_DETECTOR
177  as399xCyclicPowerRegulation();
178 #endif
179  r = as399xGetReflectedPower();
180  i = r & 0xff;
181  q = r >> 8;
182  refl = (i * i) + (q * q);
183 
184  r = refl;
185  if (refl > 0xffff) r= 0xffff;
186 
187  return r;

190 u8 tunerClimbOneParam( u8 el, u8 *elval, u16 *reflectedPower, u16 *maxSteps)
191 {
192  u16 refl, start = *reflectedPower;
193  s8 dir = 0;
194  s8 add = 0;
195  u8 improvement = 3;
196  u8 bestelval;
197  if (*maxSteps == 0)
198  {
199  return 0;
200  }
201 
202  if ( *elval != 0 )
203  {
204  tunerSetCap( el, *elval - 1 );
205  refl = tunerGetReflected();
206  if ( refl <= *reflectedPower )
207  {
208  *reflectedPower = refl;
209  dir = -1;
210  }
211  }
212 
213  if ( *elval < 31 )
214  {
215  tunerSetCap( el, *elval + 1 );
216  refl = tunerGetReflected();
217  if ( refl <= *reflectedPower )
218  {
219  *reflectedPower = refl;
220  dir = 1;
221  }
222  }
223 
224  /* if it's possible we try to change the value by 2 and check what direction is better.
225  * This improves tuning in the case of a local minima. */
226  if ( *elval > 1 )
227  {
228  tunerSetCap( el, *elval - 2 );
229  refl = tunerGetReflected();
230  if ( refl <= *reflectedPower )
231  {
232  *reflectedPower = refl;
233  dir = -1;
234  add = -1;
235  }
236  }
237 
238  if ( *elval < 30 )
239  {
240  tunerSetCap( el, *elval + 2 );
241  refl = tunerGetReflected();
242  if ( refl <= *reflectedPower )
243  {
244  *reflectedPower = refl;
245  dir = 1;
246  add = 1;
247  }
248  }
249 
250  *elval += add;
251  *elval += dir;
252  bestelval = *elval;
253 
254  if (dir!=0)
255  {
256  (*maxSteps)--;
257  while ( improvement && *maxSteps )
258  {
259  if ( *elval == 0 ) break;
260  if ( *elval == 31 ) break;
261  tunerSetCap( el, *elval + dir );
262  refl = tunerGetReflected();
263  if ( refl <= *reflectedPower )
264  {
265  (*maxSteps)--;
266  *reflectedPower = refl;
267  *elval += dir;
268  bestelval = *elval;
269  improvement = 3; /* we don't want to stop when a local minima is hit, so we
270  continue tuning for 3 more steps, even if they are worse than the one before.
271  If it does not improve in this 3 steps, we will go back to the best value. */
272  }else{
273  improvement--;
274  }
275  }
276  }
277  *elval = bestelval;
278  tunerSetCap( el, *elval);
279  return start > (*reflectedPower);
280 }

282 void tunerOneHillClimb( struct tunerParams *p, u16 maxSteps)
283 {
284  u8 improvement = 1;
285 
286  tunerSetCap(TUNER_Cout,p->cout);
287  tunerSetCap(TUNER_Clen,p->clen);
288  tunerSetCap(TUNER_Cin,p->cin);
289 
290  p->reflectedPower = tunerGetReflected();
291 
292  while (maxSteps && improvement)
293  {
294 
295  improvement = tunerClimbOneParam(TUNER_Clen, &p->clen, &p->reflectedPower, &maxSteps);
296  improvement |= tunerClimbOneParam(TUNER_Cout, &p->cout, &p->reflectedPower, &maxSteps);
297  improvement |= tunerClimbOneParam(TUNER_Cin, &p->cin, &p->reflectedPower, &maxSteps);
298 
299  }

302  tunerSetCap(TUNER_Cout, p->cout);
303  tunerSetCap(TUNER_Clen, p->clen);
